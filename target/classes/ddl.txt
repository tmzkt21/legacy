	user_no VARCHAR(10),
	rank VARCHAR(10),
	USER VARCHAR(10),
	tear VARCHAR(10),
	POINT VARCHAR(5),
	LEVEL VARCHAR(4),
	wins VARCHAR(2)
	)ENGINE=INNODB DEFAULT CHARSET=UTF8; 
	
CREATE TABLE car(
	car_no VARCHAR(10),
	img VARCHAR(50),
	NAME VARCHAR(20),
	YEAR VARCHAR(10),
	fuel VARCHAR(10),
	km VARCHAR(1),
	price VARCHAR(10),
	span VARCHAR(10)
	spa text
)ENGINE=INNODB DEFAULT CHARSET=UTF8;
